package com.github.sneha.templates.springboot_starter.config;

import org.springframework.cache.CacheManager;
import org.springframework.cache.concurrent.ConcurrentMapCacheManager;
import org.springframework.context.annotation.Bean;
import org.springframework.data.redis.cache.RedisCacheManager;
import org.springframework.data.redis.connection.RedisConnectionFactory;

public class CacheConfig {

    @Bean
    public CacheManager cacheManager(RedisConnectionFactory redisConnectionFactory) {
        // Choose either Redis, EhCache, or ConcurrentMapCache as per requirement
        return RedisCacheManager.create(redisConnectionFactory);  // Use RedisCacheManager
        // OR
        // return new ConcurrentMapCacheManager("defaultCache"); // Use in-memory cache
        // OR
        // return new EhCacheCacheManager(ehCacheManager()); // Use EhCache
    }
}
