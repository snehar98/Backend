package com.github.sneha.templates.springboot_starter.config;

import io.swagger.v3.oas.models.OpenAPI;
import io.swagger.v3.oas.models.info.Info;
import io.swagger.v3.oas.models.servers.Server;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.List;

@Configuration
public class OpenAPIConfig {

    @Bean
    public OpenAPI customOpenAPI() {
        return new OpenAPI()
                .servers(
                        List.of(
                                new Server().url("http://localhost:8080").description("Application Server"),
                                new Server().url("http://localhost:9000").description("Management Server")
                        )
                )
                .info(new Info()
                        .title("Application API")
                        .description("OpenAPI documentation for Application and Actuator")
                        .version("1.0"));
    }
}
