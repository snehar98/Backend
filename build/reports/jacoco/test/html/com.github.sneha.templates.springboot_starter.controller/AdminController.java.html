package com.github.sneha.templates.springboot_starter.controller;

import com.github.sneha.templates.springboot_starter.advice.LogExecutionTime;
import com.github.sneha.templates.springboot_starter.service.CacheEvictService;
import io.swagger.v3.oas.annotations.Operation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

/**
 *
 * @author sneharavikumartl
 *
 */

@RestController
@RequestMapping("/admin")
public class AdminController {

    @Autowired
    CacheEvictService cacheEvictService;

    @DeleteMapping("/clearAllCache")
    @Operation(summary = "Clear Cache", description = "Deletes all the caches related to the application")
    @LogExecutionTime
    public String clearRedisCache(){
        cacheEvictService.clearCache();
        return "Success";
    }

}
